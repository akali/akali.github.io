#include <bits/stdc++.h>

using namespace std;

struct rectangle {
	int x, y, x2, y2;
} r[500];

vector <int> x, y;

inline bool check(rectangle a, rectangle b) {
	return (b.x <= a.x && a.x <= b.x2) &&
				 (b.x <= a.x2 && a.x2 <= b.x2) &&
				 (b.y <= a.y && a.y <= b.y2) &&
				 (b.y <= a.y2 && a.y2 <= b.y2);
}

int main() {
	int n; scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d%d%d%d", &r[i].x, &r[i].y, &r[i].x2, &r[i].y2);
		if (r[i].x > r[i].x2)
			swap(r[i].x, r[i].x2);
		if (r[i].y > r[i].y2)
			swap(r[i].y, r[i].y2);
		x.push_back(r[i].x);
		x.push_back(r[i].x2);
		y.push_back(r[i].y);
		y.push_back(r[i].y2);
	}

	sort(x.begin(), x.end());
	x.resize(unique(x.begin(), x.end()) - x.begin());
	
	sort(y.begin(), y.end());
	y.resize(unique(y.begin(), y.end()) - y.begin());

	int res = 0;

	for (int i = 0; i + 1 < (int)x.size(); ++i)
		for (int j = 0; j + 1 < (int)y.size(); ++j)
			for (int k = 0; k < n; ++k) {
				if (check({x[i], y[j], x[i + 1], y[j + 1]}, r[k])) {
					res += (x[i + 1] - x[i]) * (y[j + 1] - y[j]);
					break;
				}
			}
	printf("%d\n", res);

	return 0;
}
