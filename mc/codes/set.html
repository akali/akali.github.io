struct item {
	int a, b;
	item() {}
	item(int a, int b) : a(a), b(b) {}
};

struct cmp {
	inline bool operator() (const item &a, const item &b) const {
		if (a.a == b.a)
			return a.b > b.b;
		return a.a > b.a;
	}
};

struct item_with_operator {
	int a, b;
	item_with_operator() {}
	item_with_operator(int a, int b) : a(a), b(b) {}
	inline bool operator < (const item_with_operator &x) const {
		if (a == x.a)
			return b < x.b;
		return a < x.a;
	}
};

set <item, cmp> q;
set <item_with_operator> w;
